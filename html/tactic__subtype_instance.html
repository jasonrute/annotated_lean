 54  /-- builds instances for algebraic substructures
 55  
 56  Example:
 57  ```lean
 58  variables {α : Type*} [monoid α] {s : set α}
 59  
 60  class is_submonoid (s : set α) : Prop :=
 61  (one_mem : (1:α) ∈ s)
 62  (mul_mem {a b} : a ∈ s → b ∈ s → a * b ∈ s)
 63  
 64  instance subtype.monoid {s : set α} [is_submonoid s] : monoid s :=
 65  by subtype_instance
 66  ````
