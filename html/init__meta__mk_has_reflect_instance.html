 46  /- Solve the subgoal for constructor `F_name` -/
 47  private meta def has_reflect_case (I_name F_name : name) (field_names : list name) : tactic unit :=

 48  do field_quotes ← mk_reflect I_name F_name field_names 0,

src                    └────────┘
typ     └──────────┘   └────────┘ └────┘ └────┘ └─────────┘
 49     -- fn should be of the form `F_name ps fs`, where ps are the inductive parameter arguments,
 50     -- and `fs.length = field_names.length`

 53     let fn := field_names.foldl (λ fn _, expr.app_fn fn) fn,

src                          └────┘          └─────────┘
