  9  /-!
 10  `bundled c` provides a uniform structure for bundling a type equipped with a type class.
 11  
 12  We provide `category` instances for these in `unbundled_hom.lean` (for categories with unbundled
 13  homs, e.g. topological spaces) and in `bundled_hom.lean` (for categories with bundled homs, e.g.
 14  monoids).
 15  -/

 17  universes u v

typ                   └──┘

 22  /-- `bundled` is a type bundled with a type class instance for that type. Only
 23  the type class is exposed as a parameter. -/
 24  structure bundled (c : Type u → Type v) : Type (max (u+1) v) :=

 37  /-
 38  `bundled.map` is reducible so that, if we define a category
 39  
 40    def Ring : Type (u+1) := induced_category SemiRing (bundled.map @ring.to_semiring)
 41  
 42  instance search is able to "see" that a morphism R ⟶ S in Ring is really
 43  a (semi)ring homomorphism from R.α to S.α, and not merely from
 44  `(bundled.map @ring.to_semiring R).α` to `(bundled.map @ring.to_semiring S).α`.
 45  -/
