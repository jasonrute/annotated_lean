 12  /-!
 13  # The natural monoidal structure on any category with finite (co)products.
 14  
 15  A category with a monoidal structure provided in this way is sometimes called a (co)cartesian category,
 16  although this is also sometimes used to mean a finitely complete category.
 17  (See <https://ncatlab.org/nlab/show/cartesian+category>.)
 18  
 19  As this works with either products or coproducts, we don't set up either construct as an instance.
 20  -/

 34  /-- A category with finite products has a natural monoidal structure. -/
 35  def monoidal_of_has_finite_products [has_finite_products.{v} C] : monoidal_category C :=
 36  { tensor_unit  := terminal C,
 37    tensor_obj   := λ X Y, limits.prod X Y,
 38    tensor_hom   := λ _ _ _ _ f g, limits.prod.map f g,
 39    associator   := prod.associator,
 40    left_unitor  := prod.left_unitor,
 41    right_unitor := prod.right_unitor,
 42    pentagon'    := prod.pentagon,
 43    triangle'    := prod.triangle,
 44    associator_naturality' := @prod.associator_naturality _ _ _, }
 45  
 46  /-- A category with finite coproducts has a natural monoidal structure. -/
 47  def monoidal_of_has_finite_coproducts [has_finite_coproducts.{v} C] : monoidal_category C :=
 48  { tensor_unit  := initial C,
 49    tensor_obj   := λ X Y, limits.coprod X Y,
 50    tensor_hom   := λ _ _ _ _ f g, limits.coprod.map f g,
 51    associator   := coprod.associator,
 52    left_unitor  := coprod.left_unitor,
 53    right_unitor := coprod.right_unitor,
 54    pentagon'    := coprod.pentagon,
 55    triangle'    := coprod.triangle,
 56    associator_naturality' := @coprod.associator_naturality _ _ _, }
 57  
 58  end

 60  end category_theory
