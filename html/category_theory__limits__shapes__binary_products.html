 10  /-!
 11  # Binary (co)products
 12  
 13  We define a category `walking_pair`, which is the index category
 14  for a binary (co)product diagram. A convenience method `pair X Y`
 15  constructs the functor from the walking pair, hitting the given objects.
 16  
 17  We define `prod X Y` and `coprod X Y` as limits and colimits of such functors.
 18  
 19  Typeclasses `has_binary_products` and `has_binary_coproducts` assert the existence
 20  of (co)limits shaped as walking pairs.
 21  -/

 60  abbreviation binary_fan (X Y : C) := cone (pair X Y)
 61  abbreviation binary_cofan (X Y : C) := cocone (pair X Y)

 65  def binary_fan.mk {P : C} (π₁ : P ⟶ X) (π₂ : P ⟶ Y) : binary_fan X Y :=

 70    ι := { app := λ j, walking_pair.cases_on j ι₁ ι₂ }}

 81  abbreviation prod (X Y : C) [has_limit (pair X Y)] := limit (pair X Y)
 82  abbreviation coprod (X Y : C) [has_colimit (pair X Y)] := colimit (pair X Y)
 83  
 84  notation X `⨯`:20 Y:20 := prod X Y
 85  notation X `⨿`:20 Y:20 := coprod X Y
 86  
 87  abbreviation prod.fst {X Y : C} [has_limit (pair X Y)] : X ⨯ Y ⟶ X :=
 88  limit.π (pair X Y) walking_pair.left
 89  abbreviation prod.snd {X Y : C} [has_limit (pair X Y)] : X ⨯ Y ⟶ Y :=
 90  limit.π (pair X Y) walking_pair.right
 91  abbreviation coprod.inl {X Y : C} [has_colimit (pair X Y)] : X ⟶ X ⨿ Y :=
 92  colimit.ι (pair X Y) walking_pair.left
 93  abbreviation coprod.inr {X Y : C} [has_colimit (pair X Y)] : Y ⟶ X ⨿ Y :=
 94  colimit.ι (pair X Y) walking_pair.right
 95  
 96  abbreviation prod.lift {W X Y : C} [has_limit (pair X Y)] (f : W ⟶ X) (g : W ⟶ Y) : W ⟶ X ⨯ Y :=
 97  limit.lift _ (binary_fan.mk f g)
 98  abbreviation coprod.desc {W X Y : C} [has_colimit (pair X Y)] (f : X ⟶ W) (g : Y ⟶ W) : X ⨿ Y ⟶ W :=
 99  colimit.desc _ (binary_cofan.mk f g)
100  
101  abbreviation prod.map {W X Y Z : C} [has_limits_of_shape.{v} (discrete walking_pair) C]
102    (f : W ⟶ Y) (g : X ⟶ Z) : W ⨯ X ⟶ Y ⨯ Z :=
103  lim.map (map_pair f g)
104  abbreviation coprod.map {W X Y Z : C} [has_colimits_of_shape.{v} (discrete walking_pair) C]
105    (f : W ⟶ Y) (g : X ⟶ Z) : W ⨿ X ⟶ Y ⨿ Z :=
106  colim.map (map_pair f g)

140  /-- The associator isomorphism for binary products. -/
141  @[simps] def prod.associator
doc    └───┘
142    (P Q R : C) : (P ⨯ Q) ⨯ R ≅ P ⨯ (Q ⨯ R) :=
143  { hom :=
144    prod.lift
145      (prod.fst ≫ prod.fst)
146      (prod.lift (prod.fst ≫ prod.snd) prod.snd),
