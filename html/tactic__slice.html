 55  namespace interactive
 56  /--
 57  `slice` is a conv tactic; if the current focus is a composition of several morphisms,
 58  `slice a b` reassociates as needed, and zooms in on the `a`-th through `b`-th morphisms.
 59  
 60  Thus if the current focus is `(a ≫ b) ≫ ((c ≫ d) ≫ e)`, then `slice 2 3` zooms to `b ≫ c`.
 61   -/
 62  meta def slice := conv.slice
id            └───┘    └────────┘
src                    └────────┘
typ           └───┘    └────────┘
 63  end interactive
 64  end conv
