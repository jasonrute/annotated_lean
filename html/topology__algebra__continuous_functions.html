doc    └─────────┘
 15  instance continuous_submonoid (α : Type u) (β : Type v) [topological_space α] [topological_space β]
 16    [monoid β] [topological_monoid β] : is_submonoid { f : α → β | continuous f } :=
 17  { one_mem := @continuous_const _ _ _ _ 1,
 18    mul_mem :=
 19    λ f g fc gc, continuous.comp (topological_monoid.continuous_mul β) (continuous.prod_mk fc gc) }.
