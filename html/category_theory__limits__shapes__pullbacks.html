 10  /-!
 11  # Pullbacks
 12  
 13  We define a category `walking_cospan` (resp. `walking_span`), which is the index category
 14  for the given data for a pullback (resp. pushout) diagram. Convenience methods `cospan f g`
 15  and `span f g` construct functors from the walking (co)span, hitting the given morphisms.
 16  
 17  We define `pullback f g` and `pushout f g` as limits and colimits of such functors.
 18  
 19  Typeclasses `has_pullbacks` and `has_pushouts` assert the existence of (co)limits shaped as
 20  walking (co)spans.
 21  -/

115  /-- `cospan f g` is the functor from the walking cospan hitting `f` and `g`. -/
116  def cospan {X Y Z : C} (f : X ⟶ Z) (g : Y ⟶ Z) : walking_cospan.{v} ⥤ C :=
117  { obj := λ x, match x with
118    | left := X
119    | right := Y
120    | one := Z
121    end,
122    map := λ x y h, match x, y, h with
123    | _, _, (id _) := 𝟙 _
124    | _, _, inl := f
125    | _, _, inr := g
126    end }
127  
128  /-- `span f g` is the functor from the walking span hitting `f` and `g`. -/
129  def span {X Y Z : C} (f : X ⟶ Y) (g : X ⟶ Z) : walking_span.{v} ⥤ C :=
130  { obj := λ x, match x with
131    | zero := X
132    | left := Y
133    | right := Z
134    end,
135    map := λ x y h, match x, y, h with
136    | _, _, (id _) := 𝟙 _
137    | _, _, fst := f
138    | _, _, snd := g
139    end }

267  /-- `pullback f g` computes the pullback of a pair of morphisms with the same target. -/
268  abbreviation pullback {X Y Z : C} (f : X ⟶ Z) (g : Y ⟶ Z) [has_limit (cospan f g)] :=
269  limit (cospan f g)
270  /-- `pushout f g` computes the pushout of a pair of morphisms with the same source. -/
271  abbreviation pushout {X Y Z : C} (f : X ⟶ Y) (g : X ⟶ Z) [has_colimit (span f g)] :=
272  colimit (span f g)
273  
274  abbreviation pullback.fst {X Y Z : C} {f : X ⟶ Z} {g : Y ⟶ Z} [has_limit (cospan f g)] : pullback f g ⟶ X :=
275  limit.π (cospan f g) walking_cospan.left
276  abbreviation pullback.snd {X Y Z : C} {f : X ⟶ Z} {g : Y ⟶ Z} [has_limit (cospan f g)] : pullback f g ⟶ Y :=
277  limit.π (cospan f g) walking_cospan.right
278  abbreviation pushout.inl {X Y Z : C} {f : X ⟶ Y} {g : X ⟶ Z} [has_colimit (span f g)] : Y ⟶ pushout f g :=
279  colimit.ι (span f g) walking_span.left
280  abbreviation pushout.inr {X Y Z : C} {f : X ⟶ Y} {g : X ⟶ Z} [has_colimit (span f g)] : Z ⟶ pushout f g :=
281  colimit.ι (span f g) walking_span.right
