 36  /-- `pi_obj f` computes the product of a family of elements `f`. (It is defined as an abbreviation
 37     for `limit (functor.of_function f)`, so for most facts about `pi_obj f`, you will just use general facts
 38     about limits.) -/
 39  abbreviation pi_obj (f : β → C) [has_limit (functor.of_function f)] := limit (functor.of_function f)
 40  /-- `sigma_obj f` computes the coproduct of a family of elements `f`. (It is defined as an abbreviation
 41     for `colimit (functor.of_function f)`, so for most facts about `sigma_obj f`, you will just use general facts
 42     about colimits.) -/
 43  abbreviation sigma_obj (f : β → C) [has_colimit (functor.of_function f)] := colimit (functor.of_function f)
 44  
 45  notation `∏ ` f:20 := pi_obj f
 46  notation `∐ ` f:20 := sigma_obj f

 48  abbreviation pi.π (f : β → C) [has_limit (functor.of_function f)] (b : β) : ∏ f ⟶ f b :=
 49  limit.π (functor.of_function f) b
 50  abbreviation sigma.ι (f : β → C) [has_colimit (functor.of_function f)] (b : β) : f b ⟶ ∐ f :=
 51  colimit.ι (functor.of_function f) b
 52  
 53  abbreviation pi.lift {f : β → C} [has_limit (functor.of_function f)] {P : C} (p : Π b, P ⟶ f b) : P ⟶ ∏ f :=
 54  limit.lift _ (fan.mk p)
 55  abbreviation sigma.desc {f : β → C} [has_colimit (functor.of_function f)] {P : C} (p : Π b, f b ⟶ P) : ∐ f ⟶ P :=
 56  colimit.desc _ (cofan.mk p)
