 11  /--  We can specify the scope of application of some tactics using
 12     the following type.
 13  
 14     - all : all occurrences of a given term are considered.
 15  
 16     - pos [1, 3] : only the first and third occurrences of a given
 17       term are consiered.
 18  
 19     - neg [2] : all but the second occurrence of a given term
 20       are considered. -/
 21  inductive occurrences
 22  | all
 23  | pos : list nat → occurrences
id           └──┘ └─┘
src          └──┘ └─┘
typ          └──┘ └─┘
 24  | neg : list nat → occurrences
id           └──┘ └─┘
src          └──┘ └─┘
typ          └──┘ └─┘
 25  
 26  open occurrences

 29  | all                  p := tt
