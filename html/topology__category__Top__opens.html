 32  /-- `opens.map f` gives the functor from open sets in Y to open set in X,
 33      given by taking preimages under f. -/
 34  def map (f : X ⟶ Y) : opens Y ⥤ opens X :=
 35  { obj := λ U, ⟨ f.val ⁻¹' U.val, f.property _ U.property ⟩,
 36    map := λ U V i, ⟨ ⟨ λ a b, i.down.down b ⟩ ⟩ }.
