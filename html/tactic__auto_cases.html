 12  meta def auto_cases_at (h : expr) : tactic string :=
id                               └──┘    └────┘ └────┘
src                              └──┘    └────┘ └────┘
typ                              └──┘    └────┘ └────┘
doc                              └──┘
 13  do t' ← infer_type h,
id      └┘   └────────┘ ┴
src          └────────┘
typ     └┘   └────────┘ ┴
doc          └────────┘
 14    t' ← whnf t',
id     └┘   └──┘ └┘
src         └──┘
typ    └┘   └──┘ └┘
doc         └──┘
 15    let use_cases := match t' with
id         └───────┘          └┘
typ        └───────┘          └┘
 16    | `(empty)     := tt
id         └───┘         └┘
src        └───┘         └┘
typ        └───┘         └┘
 17    | `(pempty)    := tt
id         └────┘        └┘
src        └────┘        └┘
typ        └────┘        └┘
doc        └────┘
 18    | `(false)     := tt
id         └───┘         └┘
src        └───┘         └┘
typ        └───┘         └┘
 19    | `(unit)      := tt
id         └──┘          └┘
src        └──┘          └┘
typ        └──┘          └┘
doc        └──┘
 20    | `(punit)     := tt
id         └───┘         └┘
src        └───┘         └┘
typ        └───┘         └┘
 21    | `(ulift _)   := tt
id         └───┘         └┘
src        └───┘         └┘
typ        └───┘         └┘
doc        └───┘
 22    | `(plift _)   := tt
id         └───┘         └┘
src        └───┘         └┘
typ        └───┘         └┘
doc        └───┘
 23    | `(prod _ _)  := tt
id         └──┘          └┘
src        └──┘          └┘
typ        └──┘          └┘
 24    | `(and _ _)   := tt
id         └─┘           └┘
src        └─┘           └┘
typ        └─┘           └┘
 25    | `(sigma _)   := tt
id         └───┘         └┘
src        └───┘         └┘
typ        └───┘         └┘
 26    | `(psigma _)  := tt
id         └────┘        └┘
src        └────┘        └┘
typ        └────┘        └┘
 27    | `(subtype _) := tt
id         └─────┘       └┘
src        └─────┘       └┘
typ        └─────┘       └┘
 28    | `(Exists _)  := tt
id         └────┘        └┘
src        └────┘        └┘
typ        └────┘        └┘
 29    | `(fin 0)     := tt
id         └─┘           └┘
src        └─┘           └┘
typ        └─┘           └┘
 30    | `(sum _ _)   := tt -- This is perhaps dangerous!
id         └─┘           └┘
src        └─┘           └┘
typ        └─┘           └┘
 31    | `(or _ _)    := tt -- This is perhaps dangerous!
id         └┘            └┘
src        └┘            └┘
typ        └┘            └┘
 32    | `(iff _ _)   := tt -- This is perhaps dangerous!
id         └─┘           └┘
src        └─┘           └┘
typ        └─┘           └┘
 33    | _            := ff
id                       └┘
src                      └┘
typ                      └┘
 34    end,
 35    if use_cases then
id        └───────┘
typ       └───────┘
 36      do cases h, pp ← pp h, return ("cases " ++ pp.to_string)
id          └───┘ ┴  └┘   └┘ ┴  └────┘           └┘ └┘└────────┘
src         └───┘         └┘    └────┘           └┘   └────────┘
typ         └───┘ ┴  └┘   └┘ ┴  └────┘           └┘ └┘└────────┘
doc         └───┘
 37    else
 38      match t' with
id             └┘
typ            └┘
 39      -- `cases` can be dangerous on `eq` and `quot`, producing mysterious errors during type checking.
 40      -- instead we attempt `induction`
 41      | `(eq _ _)        := do induction h, pp ← pp h, return ("induction " ++ pp.to_string)
id           └┘                   └───────┘ ┴  └┘   └┘ ┴  └────┘               └┘ └┘└────────┘
src          └┘                   └───────┘         └┘    └────┘               └┘   └────────┘
typ          └┘                   └───────┘ ┴  └┘   └┘ ┴  └────┘               └┘ └┘└────────┘
doc                               └───────┘
 42      | `(quot _)        := do induction h, pp ← pp h, return ("induction " ++ pp.to_string)
id           └──┘                 └───────┘ ┴  └┘   └┘ ┴  └────┘               └┘ └┘└────────┘
src                               └───────┘         └┘    └────┘               └┘   └────────┘
typ          └──┘                 └───────┘ ┴  └┘   └┘ ┴  └────┘               └┘ └┘└────────┘
doc                               └───────┘
 43      | _                := failed
id                             └────┘
src                            └────┘
typ                            └────┘
 44      end

 46  /-- Applies `cases` or `induction` on certain hypotheses. -/
 47  meta def auto_cases : tactic string :=
id                         └────┘ └────┘
src                        └────┘ └────┘
typ                        └────┘ └────┘
 48  do l ← local_context,
id      ┴   └───────────┘
src         └───────────┘
typ     ┴   └───────────┘
doc         └───────────┘
 49     results ← successes (l.reverse.map(λ h, auto_cases_at h)),
id      └─────┘   └───────┘  ┴└──────┘└──┘   ┴  └───────────┘ ┴
src               └───────┘   └──────┘└──┘      └───────────┘
typ     └─────┘   └───────┘  ┴└──────┘└──┘   ┴  └───────────┘ ┴
doc               └───────┘
 50     when (results.empty) (fail "`auto_cases` did not find any hypotheses to apply `cases` or `induction` to"),
id      └──┘  └─────┘└────┘   └──┘
src     └──┘         └────┘   └──┘
typ     └──┘  └─────┘└────┘   └──┘
 51     return (string.intercalate ", " results)
id      └────┘  └────────────────┘      └─────┘
src     └────┘  └────────────────┘
typ     └────┘  └────────────────┘      └─────┘
